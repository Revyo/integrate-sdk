## Installation

Install the Integrate SDK in your Fastify project:

```bash
bun add integrate-sdk
```

## Setup

### 1. Server Configuration

Create a server configuration file with your OAuth credentials:

```typescript
// lib/integrate-server.ts
import {
  createMCPServer,
  githubIntegration,
  gmailIntegration,
} from "integrate-sdk/server";

export const { client: serverClient, handler } = createMCPServer({
  integrations: [
    githubIntegration({
      clientId: process.env.GITHUB_CLIENT_ID!,
      clientSecret: process.env.GITHUB_CLIENT_SECRET!,
      scopes: ["repo", "user"],
    }),
    gmailIntegration({
      clientId: process.env.GMAIL_CLIENT_ID!,
      clientSecret: process.env.GMAIL_CLIENT_SECRET!,
      scopes: ["gmail.send", "gmail.readonly"],
    }),
  ],
});
```

### 2. Mount the Handler

Add OAuth route handlers to your Fastify application:

```typescript
import Fastify from "fastify";
import { handler } from "./lib/integrate-server";

const fastify = Fastify({ logger: true });

// Register OAuth catch-all route
fastify.route({
  method: ["GET", "POST"],
  url: "/api/integrate/*",
  handler: async (request, reply) => {
    // Construct Web Request from Fastify request
    const url = new URL(request.url, `http://${request.headers.host}`);
    const webRequest = new Request(url, {
      method: request.method,
      headers: request.headers as any,
      body: request.method !== "GET" ? JSON.stringify(request.body) : undefined,
    });

    const response = await handler(webRequest);

    // Forward response to Fastify
    reply.status(response.status);
    response.headers.forEach((value, key) => {
      reply.header(key, value);
    });
    return reply.send(await response.text());
  },
});

fastify.listen({ port: 3000 }, (err) => {
  if (err) throw err;
});
```

## Configuration

Add your OAuth credentials to your `.env` file:

```bash
GITHUB_CLIENT_ID=your_github_client_id
GITHUB_CLIENT_SECRET=your_github_client_secret
```

## CORS Setup

Configure CORS using Fastify's CORS integration:

```typescript
import fastifyCors from "@fastify/cors";

fastify.register(fastifyCors, {
  origin: "http://localhost:3001",
  credentials: true,
});
```

## Usage Examples

### Client-Side Authorization

```typescript
import { createMCPClient, githubIntegration } from "integrate-sdk";

const client = createMCPClient({
  integrations: [
    githubIntegration({
      scopes: ["repo", "user"],
    }),
  ],
});

// Start OAuth flow
await client.authorize("github");
```
